44string huffman::decToBin(int inNum) {
45 //to convert decimal to binary
46 string temp = "", res = "";
47 while (inNum > 0) {
48 temp += (inNum % 2 + '0');
49 inNum /= 2;
50 }
51 res.append(8 - temp.length(), '0');
52 for (int i = temp.length() - 1; i >= 0; i--) {
53 res += temp[i];
54 }
55 return res;

96void huffman::createTree() {
97 //Creating Huffman Tree with the Min Heap created earlier
98 Node *left, *right;
99 priority_queue <Node*, vector<Node*>, Compare> tempPQ(minHeap);
100 while (tempPQ.size() != 1)
101 {
102 left = tempPQ.top();
103 tempPQ.pop();
104
105 right = tempPQ.top();
106 tempPQ.pop();
107
108 root = new Node();
109 root->freq = left->freq + right->freq;
110
111 root->left = left;
112 root->right = right;
113 tempPQ.push(root);
114 }
115}

122void huffman::saveEncodedFile() {
123 //Saving encoded (.huf) file
124 inFile.open(inFileName, ios::in);
125 outFile.open(outFileName, ios::out | ios::binary);
126 string in = "";
127 string s = "";
128 char id;
129
130 //Saving the meta data (huffman tree)
131 in += (char)minHeap.size();
132 priority_queue <Node*, vector<Node*>, Compare> tempPQ(minHeap);
133 while (!tempPQ.empty()) {
134 Node* curr = tempPQ.top();
135 in += curr->data;
136 //Saving 16 decimal values representing code of curr->data
137 s.assign(127 - curr->code.length(), '0');
138 s += '1';
139 s += curr->code;
140 //Saving decimal values of every 8-bit binary code
141 in += (char)binToDec(s.substr(0, 8));
142 for (int i = 0; i < 15; i++) {
143 s = s.substr(8);
144 in += (char)binToDec(s.substr(0, 8));
145 }
146 tempPQ.pop();
147 }
148 s.clear();
149
150 //Saving codes of every character appearing in the input file
151 inFile.get(id);
152 while (!inFile.eof()) {
153 s += arr[id]->code;
154 //Saving decimal values of every 8-bit binary code
155 while (s.length() > 8) {
156 in += (char)binToDec(s.substr(0, 8));
157 s = s.substr(8);
158 }
159 inFile.get(id);
160 }
161
162 //Finally if bits remaining are less than 8, append 0's
163 int count = 8 - s.length();
164 if (s.length() < 8) {
165 s.append(count, '0');
166 }
167 in += (char)binToDec(s);
168 //append count of appended 0's
169 in += (char)count;
170
171 //write the in string to the output file
172 outFile.write(in.c_str(), in.size());
173 inFile.close();
174 outFile.close();
175}

177void huffman::saveDecodedFile() {
178 inFile.open(inFileName, ios::in | ios::binary);
179 outFile.open(outFileName, ios::out);
180 unsigned char size;
181 inFile.read(reinterpret_cast<char*>(&size), 1);
182 //Reading count at the end of the file which is number of bits appended to make final value 8-bit
183 inFile.seekg(-1, ios::end);
184 char count0;
185 inFile.read(&count0, 1);
186 //Ignoring the meta data (huffman tree) (1 + 17 * size) and reading remaining file
187 inFile.seekg(1 + 17 * size, ios::beg);
188
189 vector<unsigned char> text;
190 unsigned char textseg;
191 inFile.read(reinterpret_cast<char*>(&textseg), 1);
192 while (!inFile.eof()) {
193 text.push_back(textseg);
194 inFile.read(reinterpret_cast<char*>(&textseg), 1);
195 }
196
197 Node *curr = root;
198 string path;
199 for (int i = 0; i < text.size() - 1; i++) {
200 //Converting decimal number to its equivalent 8-bit binary code
201 path = decToBin(text[i]);
202 if (i == text.size() - 2) {
203 path = path.substr(0, 8 - count0);
204 }
205 //Traversing huffman tree and appending resultant data to the file
206 for (int j = 0; j < path.size(); j++) {
207 if (path[j] == '0') {
208 curr = curr->left;
209 }
210 else {
211 curr = curr->right;
212 }
213
214 if (curr->left == NULL && curr->right == NULL) {
215 outFile.put(curr->data);
216 curr = root;
217 }
218 }
219 }
220 inFile.close();
221 outFile.close();
222}

224void huffman::getTree() {
225 inFile.open(inFileName, ios::in | ios::binary);
226 //Reading size of MinHeap
227 unsigned char size;
228 inFile.read(reinterpret_cast<char*>(&size), 1);
229 root = new Node();
230 //next size * (1 + 16) characters contain (char)data and (string)code[in decimal]
231 for(int i = 0; i < size; i++) {
232 char aCode;
233 unsigned char hCodeC[16];
234 inFile.read(&aCode, 1);
235 inFile.read(reinterpret_cast<char*>(hCodeC), 16);
236 //converting decimal characters into their binary equivalent to obtain code
237 string hCodeStr = "";
238 for (int i = 0; i < 16; i++) {
239 hCodeStr += decToBin(hCodeC[i]);
240 }
241 //Removing padding by ignoring first (127 - curr->code.length()) '0's and next '1' character
242 int j = 0;
243 while (hCodeStr[j] == '0') {
244 j++;
245 }
246 hCodeStr = hCodeStr.substr(j+1);
247 //Adding node with aCode data and hCodeStr string to the huffman tree
248 buildTree(aCode, hCodeStr);
249 }
250 inFile.close();
251}
